86 template <
87 typename StringViewLike,
88 typename BufferSequence,
89 std::enable_if_t<
90 std::is_convertible_v<std::decay_t<StringViewLike>, std::string_view> &&
91 (
92 is_buffer_sequence<std::decay_t<BufferSequence>>::value ||
93 std::is_convertible_v<std::decay_t<BufferSequence>, std::string_view>
94 ),
95 std::nullptr_t
96 > = nullptr
97 >
98 basic_publish_packet(
99 packet_id_t packet_id,
100 StringViewLike&& topic_name,
101 BufferSequence&& payloads,
102 pub::opts pubopts,
103 properties props = {}
104 )
105 : fixed_header_(
106 make_fixed_header(control_packet_type::publish, 0b0000) | std::uint8_t(pubopts)
107 ),
108 topic_name_{std::string{std::forward<StringViewLike>(topic_name)}},
109 packet_id_(PacketIdBytes),
110 property_length_(async_mqtt::size(props)),
111 props_(force_move(props)),
112 remaining_length_(
113 2 // topic name length
114 + topic_name_.size() // topic name
115 + ( (pubopts.get_qos() == qos::at_least_once || pubopts.get_qos() == qos::exactly_once)
116 ? PacketIdBytes // packet_id
117 : 0)
118 )
119 {
120 using namespace std::literals;
121 topic_name_length_buf_.resize(topic_name_length_buf_.capacity());
122 endian_store(
123 boost::numeric_cast<std::uint16_t>(topic_name_.size()),
124 topic_name_length_buf_.data()
125 );
126
127 if constexpr (std::is_convertible_v<std::decay_t<BufferSequence>, std::string_view>) {
128 remaining_length_ += std::string_view(payloads).size();
129 payloads_.emplace_back(buffer{std::string{payloads}});
130 }
131 else {
132 auto b = buffer_sequence_begin(payloads);
133 auto e = buffer_sequence_end(payloads);
134 auto num_of_payloads = static_cast<std::size_t>(std::distance(b, e));
135 payloads_.reserve(num_of_payloads);
136 for (; b != e; ++b) {
137 auto const& payload = *b;
138 remaining_length_ += payload.size();
139 payloads_.push_back(payload);
140 }
141 }
142
143 if (!utf8string_check(topic_name_)) {
144 throw make_error(
145 errc::bad_message,
146 "v5::publish_packet topic name invalid utf8"
147 );
148 }
149
150 auto pb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(property_length_));
151 for (auto e : pb) {
152 property_length_buf_.push_back(e);
153 }
154
155 for (auto const& prop : props_) {
156 auto id = prop.id();
157 if (!validate_property(property_location::publish, id)) {
158 throw make_error(
159 errc::bad_message,
160 "v5::publish_packet property "s + id_to_str(id) + " is not allowed"
161 );
162 }
163 }
164
165 remaining_length_ += property_length_buf_.size() + property_length_;
166
167 auto rb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
168 for (auto e : rb) {
169 remaining_length_buf_.push_back(e);
170 }
171
172 if (pubopts.get_qos() == qos::at_least_once ||
173 pubopts.get_qos() == qos::exactly_once) {
174 if (packet_id == 0) {
175 throw make_error(
176 errc::bad_message,
177 "v5::publish_packet qos not 0 but packet_id is 0"
178 );
179 }
180 endian_store(packet_id, packet_id_.data());
181 }
182 else {
183 if (packet_id != 0) {
184 throw make_error(
185 errc::bad_message,
186 "v5::publish_packet qos0 but non 0 packet_id"
187 );
188 }
189 endian_store(0, packet_id_.data());
190 }
191 }

244 std::vector<as::const_buffer> const_buffer_sequence() const {
245 std::vector<as::const_buffer> ret;
246 ret.reserve(num_of_const_buffer_sequence());
247 ret.emplace_back(as::buffer(&fixed_header_, 1));
248 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
249 ret.emplace_back(as::buffer(topic_name_length_buf_.data(), topic_name_length_buf_.size()));
250 ret.emplace_back(as::buffer(topic_name_));
251 if (packet_id() != 0) {
252 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
253 }
254 ret.emplace_back(as::buffer(property_length_buf_.data(), property_length_buf_.size()));
255 auto props_cbs = async_mqtt::const_buffer_sequence(props_);
256 std::move(props_cbs.begin(), props_cbs.end(), std::back_inserter(ret));
257 for (auto const& payload : payloads_) {
258 ret.emplace_back(as::buffer(payload));
259 }
260 return ret;
261 }

544 // private constructor for internal use
545 basic_publish_packet(buffer buf)
546 : packet_id_(PacketIdBytes) {
547 // fixed_header
548 if (buf.empty()) {
549 throw make_error(
550 errc::bad_message,
551 "v5::publish_packet fixed_header doesn't exist"
552 );
553 }
554 fixed_header_ = static_cast<std::uint8_t>(buf.front());
555 buf.remove_prefix(1);
556
557 // remaining_length
558 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
559 remaining_length_ = *vl_opt;
560 }
561 else {
562 throw make_error(errc::bad_message, "v5::publish_packet remaining length is invalid");
563 }
564
565 // topic_name_length
566 if (!insert_advance(buf, topic_name_length_buf_)) {
567 throw make_error(
568 errc::bad_message,
569 "v5::publish_packet length of topic_name is invalid"
570 );
571 }
572 auto topic_name_length = endian_load<std::uint16_t>(topic_name_length_buf_.data());
573
574 // topic_name
575 if (buf.size() < topic_name_length) {
576 throw make_error(
577 errc::bad_message,
578 "v5::publish_packet topic_name doesn't match its length"
579 );
580 }
581 topic_name_ = buf.substr(0, topic_name_length);
582
583 if (!utf8string_check(topic_name_)) {
584 throw make_error(
585 errc::bad_message,
586 "v5::publish_packet topic name invalid utf8"
587 );
588 }
589
590 buf.remove_prefix(topic_name_length);
591
592 // packet_id
593 switch (pub::get_qos(fixed_header_)) {
594 case qos::at_most_once:
595 endian_store(packet_id_t{0}, packet_id_.data());
596 break;
597 case qos::at_least_once:
598 case qos::exactly_once:
599 if (!copy_advance(buf, packet_id_)) {
600 throw make_error(
601 errc::bad_message,
602 "v5::publish_packet packet_id doesn't exist"
603 );
604 }
605 break;
606 default:
607 throw make_error(
608 errc::bad_message,
609 "v5::publish_packet qos is invalid"
610 );
611 break;
612 };
613
614 // property
615 auto it = buf.begin();
616 if (auto pl_opt = variable_bytes_to_val(it, buf.end())) {
617 property_length_ = *pl_opt;
618 std::copy(buf.begin(), it, std::back_inserter(property_length_buf_));
619 buf.remove_prefix(std::size_t(std::distance(buf.begin(), it)));
620 if (buf.size() < property_length_) {
621 throw make_error(
622 errc::bad_message,
623 "v5::publish_packet properties_don't match its length"
624 );
625 }
626 auto prop_buf = buf.substr(0, property_length_);
627 props_ = make_properties(prop_buf, property_location::publish);
628 buf.remove_prefix(property_length_);
629 }
630 else {
631 throw make_error(
632 errc::bad_message,
633 "v5::publish_packet property_length is invalid"
634 );
635 }
636
637 // payload
638 if (!buf.empty()) {
639 payloads_.emplace_back(force_move(buf));
640 }
641 }
642private:
643 std::uint8_t fixed_header_;
644 buffer topic_name_;
645 static_vector<char, 2> topic_name_length_buf_;
646 static_vector<char, PacketIdBytes> packet_id_;
647 std::size_t property_length_;
648 static_vector<char, 4> property_length_buf_;
649 properties props_;
650 std::vector<buffer> payloads_;
651 std::size_t remaining_length_;
652 static_vector<char, 4> remaining_length_buf_;
653};
