126 connack_packet(buffer buf) {
127 // fixed_header
128 if (buf.empty()) {
129 throw make_error(
130 errc::bad_message,
131 "v3_1_1::connack_packet fixed_header doesn't exist"
132 );
133 }
134 all_.push_back(buf.front());
135 buf.remove_prefix(1);
136 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(all_.back()));
137 if (!cpt_opt || *cpt_opt != control_packet_type::connack) {
138 throw make_error(
139 errc::bad_message,
140 "v3_1_1::connack_packet fixed_header is invalid"
141 );
142 }
143
144 // remaining_length
145 if (buf.empty()) {
146 throw make_error(
147 errc::bad_message,
148 "v3_1_1::connack_packet remaining_length doesn't exist"
149 );
150 }
151 all_.push_back(buf.front());
152 buf.remove_prefix(1);
153 if (static_cast<std::uint8_t>(all_.back()) != 0b00000010) {
154 throw make_error(
155 errc::bad_message,
156 "v3_1_1::connack_packet remaining_length is invalid"
157 );
158 }
159
160 // variable header
161 if (buf.size() != 2) {
162 throw make_error(
163 errc::bad_message,
164 "v3_1_1::connack_packet variable header doesn't match its length"
165 );
166 }
167 all_.push_back(buf.front());
168 buf.remove_prefix(1);
169 if ((static_cast<std::uint8_t>(all_.back()) & 0b11111110)!= 0) {
170 throw make_error(
171 errc::bad_message,
172 "v3_1_1::connack_packet connect acknowledge flags is invalid"
173 );
174 }
175 all_.push_back(buf.front());
176 if (static_cast<std::uint8_t>(all_.back()) > 5) {
177 throw make_error(
178 errc::bad_message,
179 "v3_1_1::connack_packet connect_return_code is invalid"
180 );
181 }
182 }
183
184private:
185 static_vector<char, 4> all_;
