100template <std::size_t PacketIdBytes>
101inline
102basic_suback_packet<PacketIdBytes>::basic_suback_packet(buffer buf) {
103 // fixed_header
104 if (buf.empty()) {
105 throw make_error(
106 errc::bad_message,
107 "v3_1_1::suback_packet fixed_header doesn't exist"
108 );
109 }
110 fixed_header_ = static_cast<std::uint8_t>(buf.front());
111 buf.remove_prefix(1);
112 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
113 if (!cpt_opt || *cpt_opt != control_packet_type::suback) {
114 throw make_error(
115 errc::bad_message,
116 "v3_1_1::suback_packet fixed_header is invalid"
117 );
118 }
119
120 // remaining_length
121 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
122 remaining_length_ = *vl_opt;
123 }
124 else {
125 throw make_error(errc::bad_message, "v3_1_1::suback_packet remaining length is invalid");
126 }
127 if (remaining_length_ != buf.size()) {
128 throw make_error(errc::bad_message, "v3_1_1::suback_packet remaining length doesn't match buf.size()");
129 }
130
131 // packet_id
132 if (!copy_advance(buf, packet_id_)) {
133 throw make_error(
134 errc::bad_message,
135 "v3_1_1::suback_packet packet_id doesn't exist"
136 );
137 }
138
139 if (remaining_length_ == 0) {
140 throw make_error(errc::bad_message, "v3_1_1::suback_packet doesn't have entries");
141 }
142
143 while (!buf.empty()) {
144 // suback_return_code
145 if (buf.empty()) {
146 throw make_error(
147 errc::bad_message,
148 "v3_1_1::suback_packet suback_return_code doesn't exist"
149 );
150 }
151 auto rc = static_cast<suback_return_code>(buf.front());
152 entries_.emplace_back(rc);
153 buf.remove_prefix(1);
154 }
155}
