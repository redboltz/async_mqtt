50template <std::size_t PacketIdBytes>
51class basic_subscribe_packet {
52public:
53 using packet_id_t = typename packet_id_type<PacketIdBytes>::type;

61 basic_subscribe_packet(
62 packet_id_t packet_id,
63 std::vector<topic_subopts> params
64 )
65 : fixed_header_{make_fixed_header(control_packet_type::subscribe, 0b0010)},
66 entries_{force_move(params)},
67 remaining_length_{PacketIdBytes}
68 {
69 topic_length_buf_entries_.reserve(entries_.size());
70 for (auto const& e : entries_) {
71 topic_length_buf_entries_.push_back(
72 endian_static_vector(
73 boost::numeric_cast<std::uint16_t>(e.all_topic().size())
74 )
75 );
76 }
77
78 endian_store(packet_id, packet_id_.data());
79
80 for (auto const& e : entries_) {
81 // reserved bits check
82 if (static_cast<std::uint8_t>(e.opts()) & 0b11111100) {
83 throw make_error(
84 errc::bad_message,
85 "v3_1_1::subscribe_packet subopts is invalid"
86 );
87 }
88 switch (e.opts().get_qos()) {
89 case qos::at_most_once:
90 case qos::at_least_once:
91 case qos::exactly_once:
92 break;
93 default:
94 throw make_error(
95 errc::bad_message,
96 "v3_1_1::subscribe_packet qos is invalid"
97 );
98 break;
99 }
100
101 auto size = e.all_topic().size();
102 if (size > 0xffff) {
103 throw make_error(
104 errc::bad_message,
105 "v3_1_1::subscribe_packet length of topic is invalid"
106 );
107 }
108 remaining_length_ +=
109 2 + // topic filter length
110 size + // topic filter
111 1; // opts
112
113 if (!utf8string_check(e.all_topic())) {
114 throw make_error(
115 errc::bad_message,
116 "v3_1_1::subscribe_packet topic filter invalid utf8"
117 );
118 }
119 }
120
121 remaining_length_buf_ = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
122 }

128 constexpr control_packet_type type() const {
129 return control_packet_type::subscribe;

137 std::vector<as::const_buffer> const_buffer_sequence() const {
138 std::vector<as::const_buffer> ret;
139 ret.reserve(num_of_const_buffer_sequence());
140
141 ret.emplace_back(as::buffer(&fixed_header_, 1));
142
143 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
144
145 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
146
147 BOOST_ASSERT(entries_.size() == topic_length_buf_entries_.size());
148 auto it = topic_length_buf_entries_.begin();
149 for (auto const& e : entries_) {
150 ret.emplace_back(as::buffer(it->data(), it->size()));
151 ret.emplace_back(as::buffer(e.all_topic()));
152 ret.emplace_back(as::buffer(&e.opts(), 1));
153 ++it;
154 }
155
156 return ret;
157 }

209 // private constructor for internal use
210 basic_subscribe_packet(buffer buf) {
211 // fixed_header
212 if (buf.empty()) {
213 throw make_error(
214 errc::bad_message,
215 "v3_1_1::subscribe_packet fixed_header doesn't exist"
216 );
217 }
218 fixed_header_ = static_cast<std::uint8_t>(buf.front());
219 buf.remove_prefix(1);
220 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
221 if (!cpt_opt || *cpt_opt != control_packet_type::subscribe) {
222 throw make_error(
223 errc::bad_message,
224 "v3_1_1::subscribe_packet fixed_header is invalid"
225 );
226 }
227
228 // remaining_length
229 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
230 remaining_length_ = *vl_opt;
231 }
232 else {
233 throw make_error(errc::bad_message, "v3_1_1::subscribe_packet remaining length is invalid");
234 }
235 if (remaining_length_ != buf.size()) {
236 throw make_error(errc::bad_message, "v3_1_1::subscribe_packet remaining length doesn't match buf.size()");
237 }
238
239 // packet_id
240 if (!copy_advance(buf, packet_id_)) {
241 throw make_error(
242 errc::bad_message,
243 "v3_1_1::subscribe_packet packet_id doesn't exist"
244 );
245 }
246
247 if (remaining_length_ == 0) {
248 throw make_error(errc::bad_message, "v3_1_1::subscribe_packet doesn't have entries");
249 }
250
251 while (!buf.empty()) {
252 // topic_length
253 static_vector<char, 2> topic_length_buf;
254 if (!insert_advance(buf, topic_length_buf)) {
255 throw make_error(
256 errc::bad_message,
257 "v3_1_1::subscribe_packet length of topic is invalid"
258 );
259 }
260 auto topic_length = endian_load<std::uint16_t>(topic_length_buf.data());
261 topic_length_buf_entries_.push_back(topic_length_buf);
262
263 // topic
264 if (buf.size() < topic_length) {
265 throw make_error(
266 errc::bad_message,
267 "v3_1_1::subscribe_packet topic doesn't match its length"
268 );
269 }
270 auto topic = buf.substr(0, topic_length);
271
272 if (!utf8string_check(topic)) {
273 throw make_error(
274 errc::bad_message,
275 "v3_1_1::subscribe_packet topic filter invalid utf8"
276 );
277 }
278
279 buf.remove_prefix(topic_length);
280
281 // opts
282 if (buf.empty()) {
283 throw make_error(
284 errc::bad_message,
285 "v3_1_1::subscribe_packet subscribe options doesn't exist"
286 );
287 }
288 auto opts = static_cast<sub::opts>(std::uint8_t(buf.front()));
289 if (static_cast<std::uint8_t>(opts) & 0b11111100) {
290 throw make_error(
291 errc::bad_message,
292 "v3_1_1::subscribe_packet subopts is invalid"
293 );
294 }
295 switch (opts.get_qos()) {
296 case qos::at_most_once:
297 case qos::at_least_once:
298 case qos::exactly_once:
299 break;
300 default:
301 throw make_error(
302 errc::bad_message,
303 "v3_1_1::subscribe_packet qos is invalid"
304 );
305 break;
306 }
307 entries_.emplace_back(std::string{topic}, opts);
308 buf.remove_prefix(1);
309 }
310 }
311
312private:
313 std::uint8_t fixed_header_;
314 std::vector<static_vector<char, 2>> topic_length_buf_entries_;
315 std::vector<topic_subopts> entries_;
316 static_vector<char, PacketIdBytes> packet_id_ = static_vector<char, PacketIdBytes>(PacketIdBytes);
317 std::size_t remaining_length_;
318 static_vector<char, 4> remaining_length_buf_;
319};
