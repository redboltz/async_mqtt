114 }
115 all_.push_back(buf.front());
116 buf.remove_prefix(1);
117 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(all_.back()));
118 if (!cpt_opt || *cpt_opt != control_packet_type::pingreq) {
119 throw make_error(
120 errc::bad_message,
121 "v3_1_1::pingreq_packet fixed_header is invalid"
122 );
123 }
124
125 // remaining_length
126 if (buf.empty()) {
127 throw make_error(
128 errc::bad_message,
129 "v3_1_1::pingreq_packet remaining_length doesn't exist"
130 );
131 }
132 all_.push_back(buf.front());
133
134 if (static_cast<std::uint8_t>(all_.back()) != 0) {
135 throw make_error(
136 errc::bad_message,
137 "v3_1_1::pingreq_packet remaining_length is invalid"
138 );
139 }
140 }
