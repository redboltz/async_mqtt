37template <std::size_t PacketIdBytes>
38class basic_unsubscribe_packet {
39public:
40 using packet_id_t = typename packet_id_type<PacketIdBytes>::type;

50 basic_unsubscribe_packet(
51 packet_id_t packet_id,
52 std::vector<topic_sharename> params,
53 properties props = {}
54 )
55 : fixed_header_{make_fixed_header(control_packet_type::unsubscribe, 0b0010)},
56 entries_{force_move(params)},
57 remaining_length_{PacketIdBytes},
58 property_length_(async_mqtt::size(props)),
59 props_(force_move(props))
60 {
61 using namespace std::literals;
62 topic_length_buf_entries_.reserve(entries_.size());
63 for (auto const& e : entries_) {
64 topic_length_buf_entries_.push_back(
65 endian_static_vector(
66 boost::numeric_cast<std::uint16_t>(e.all_topic().size())
67 )
68 );
69 }
70
71 endian_store(packet_id, packet_id_.data());
72
73 auto pb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(property_length_));
74 for (auto e : pb) {
75 property_length_buf_.push_back(e);
76 }
77
78 for (auto const& prop : props_) {
79 auto id = prop.id();
80 if (!validate_property(property_location::unsubscribe, id)) {
81 throw make_error(
82 errc::bad_message,
83 "v5::unsubscribe_packet property "s + id_to_str(id) + " is not allowed"
84 );
85 }
86 }
87
88 remaining_length_ += property_length_buf_.size() + property_length_;
89
90 for (auto const& e : entries_) {
91 auto size = e.all_topic().size();
92 if (size > 0xffff) {
93 throw make_error(
94 errc::bad_message,
95 "v5::unsubscribe_packet length of topic is invalid"
96 );
97 }
98 remaining_length_ +=
99 2 + // topic filter length
100 size; // topic filter
101
102#if 0 // TBD
103 utf8string_check(e.all_topic());
104#endif
105 }
106
107 remaining_length_buf_ = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
108 }

110 basic_unsubscribe_packet(buffer buf) {
111 // fixed_header
112 if (buf.empty()) {
113 throw make_error(
114 errc::bad_message,
115 "v5::unsubscribe_packet fixed_header doesn't exist"
116 );
117 }
118 fixed_header_ = static_cast<std::uint8_t>(buf.front());
119 buf.remove_prefix(1);
120 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
121 if (!cpt_opt || *cpt_opt != control_packet_type::unsubscribe) {
122 throw make_error(
123 errc::bad_message,
124 "v5::unsubscribe_packet fixed_header is invalid"
125 );
126 }
127
128 // remaining_length
129 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
130 remaining_length_ = *vl_opt;
131 }
132 else {
133 throw make_error(errc::bad_message, "v5::unsubscribe_packet remaining length is invalid");
134 }
135 if (remaining_length_ != buf.size()) {
136 throw make_error(errc::bad_message, "v5::unsubscribe_packet remaining length doesn't match buf.size()");
137 }
138
139 // packet_id
140 if (!copy_advance(buf, packet_id_)) {
141 throw make_error(
142 errc::bad_message,
143 "v5::unsubscribe_packet packet_id doesn't exist"
144 );
145 }
146
147 // property
148 auto it = buf.begin();
149 if (auto pl_opt = variable_bytes_to_val(it, buf.end())) {
150 property_length_ = *pl_opt;
151 std::copy(buf.begin(), it, std::back_inserter(property_length_buf_));
152 buf.remove_prefix(std::size_t(std::distance(buf.begin(), it)));
153 if (buf.size() < property_length_) {
154 throw make_error(
155 errc::bad_message,
156 "v5::unsubscribe_packet properties_don't match its length"
157 );
158 }
159 auto prop_buf = buf.substr(0, property_length_);
160 props_ = make_properties(prop_buf, property_location::unsubscribe);
161 buf.remove_prefix(property_length_);
162 }
163 else {
164 throw make_error(
165 errc::bad_message,
166 "v5::unsubscribe_packet property_length is invalid"
167 );
168 }
169
170 if (remaining_length_ == 0) {
171 throw make_error(errc::bad_message, "v5::unsubscribe_packet doesn't have entries");
172 }
173
174 while (!buf.empty()) {
175 // topic_length
176 static_vector<char, 2> topic_length_buf;
177 if (!insert_advance(buf, topic_length_buf)) {
178 throw make_error(
179 errc::bad_message,
180 "v5::unsubscribe_packet length of topic is invalid"
181 );
182 }
183 auto topic_length = endian_load<std::uint16_t>(topic_length_buf.data());
184 topic_length_buf_entries_.push_back(topic_length_buf);
185
186 // topic
187 if (buf.size() < topic_length) {
188 throw make_error(
189 errc::bad_message,
190 "v5::unsubscribe_packet topic doesn't match its length"
191 );
192 }
193 auto topic = buf.substr(0, topic_length);
194 buf.remove_prefix(topic_length);
195 entries_.emplace_back(force_move(topic));
196 }
197 }

208 std::vector<as::const_buffer> const_buffer_sequence() const {
209 std::vector<as::const_buffer> ret;
210 ret.reserve(num_of_const_buffer_sequence());
211
212 ret.emplace_back(as::buffer(&fixed_header_, 1));
213
214 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
215
216 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
217
218 ret.emplace_back(as::buffer(property_length_buf_.data(), property_length_buf_.size()));
219 auto props_cbs = async_mqtt::const_buffer_sequence(props_);
220 std::move(props_cbs.begin(), props_cbs.end(), std::back_inserter(ret));
221
222 BOOST_ASSERT(entries_.size() == topic_length_buf_entries_.size());
223 auto it = topic_length_buf_entries_.begin();
224 for (auto const& e : entries_) {
225 ret.emplace_back(as::buffer(it->data(), it->size()));
226 ret.emplace_back(as::buffer(e.all_topic()));
227 ++it;
228 }
229
230 return ret;
231 }

248 std::size_t num_of_const_buffer_sequence() const {
249 return
250 1 + // fixed header
251 1 + // remaining length
252 1 + // packet id
253 [&] () -> std::size_t {
254 if (property_length_buf_.size() == 0) return 0;
255 return
256 1 + // property length
257 async_mqtt::num_of_const_buffer_sequence(props_);
258 }() +
259 entries_.size() * 2; // topic name length, topic name
260 }
