39template <std::size_t PacketIdBytes>
40class basic_puback_packet {
41public:
42 using packet_id_t = typename packet_id_type<PacketIdBytes>::type;

48 basic_puback_packet(
49 packet_id_t packet_id
50 )
51 : all_(all_.capacity())
52 {
53 all_[0] = static_cast<char>(make_fixed_header(control_packet_type::puback, 0b0000));
54 all_[1] = boost::numeric_cast<char>(PacketIdBytes);
55 endian_store(packet_id, &all_[2]);
56 }

58 basic_puback_packet(buffer buf) {
59 // fixed_header
60 if (buf.empty()) {
61 throw make_error(
62 errc::bad_message,
63 "v3_1_1::puback_packet fixed_header doesn't exist"
64 );
65 }
66 all_.push_back(buf.front());
67 buf.remove_prefix(1);
68 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(all_.back()));
69 if (!cpt_opt || *cpt_opt != control_packet_type::puback) {
70 throw make_error(
71 errc::bad_message,
72 "v3_1_1::puback_packet fixed_header is invalid"
73 );
74 }
75
76 // remaining_length
77 if (buf.empty()) {
78 throw make_error(
79 errc::bad_message,
80 "v3_1_1::puback_packet remaining_length doesn't exist"
81 );
82 }
83 all_.push_back(buf.front());
84 buf.remove_prefix(1);
85 if (static_cast<std::uint8_t>(all_.back()) != PacketIdBytes) {
86 throw make_error(
87 errc::bad_message,
88 "v3_1_1::puback_packet remaining_length is invalid"
89 );
90 }
91
92 // variable header
93 if (buf.size() != PacketIdBytes) {
94 throw make_error(
95 errc::bad_message,
96 "v3_1_1::puback_packet variable header doesn't match its length"
97 );
98 }
99 std::copy(buf.begin(), buf.end(), std::back_inserter(all_));
100 }
