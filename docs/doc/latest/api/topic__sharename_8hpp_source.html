32 topic_sharename(
33 AllTopic&& all_topic
34 ): all_topic_{std::forward<AllTopic>(all_topic)} {
35 auto const shared_prefix = std::string_view("$share/");
36 if (all_topic_.substr(0, shared_prefix.size()) == shared_prefix) {
37 // must have sharename
38 sharename_ = all_topic_.substr(shared_prefix.size());
39
40 auto const idx = sharename_.find_first_of('/');
41 if (idx == 0 || idx == std::string_view::npos) return;
42
43 topic_ = sharename_.substr(idx + 1);
44 sharename_ = sharename_.substr(0, sharename_.size() - topic_.size() - 1);
45 }
46 else {
47 topic_ = all_topic_;
48 }
49 }
