34 topic_sharename(
35 AllTopic&& all_topic
36 ): all_topic_{std::forward<AllTopic>(all_topic)} {
37 BOOST_ASSERT(all_topic_.size() <= 0xffff);
38 auto const shared_prefix = std::string_view("$share/");
39 if (all_topic_.substr(0, shared_prefix.size()) == shared_prefix) {
40 // must have sharename
41 sharename_ = all_topic_.substr(shared_prefix.size());
42
43 auto const idx = sharename_.find_first_of('/');
44 if (idx == 0 || idx == std::string_view::npos) return;
45
46 topic_ = sharename_.substr(idx + 1);
47 sharename_ = sharename_.substr(0, sharename_.size() - topic_.size() - 1);
48 }
49 else {
50 topic_ = all_topic_;
51 }
52 }
