911 template <typename Packet>
912 struct send_impl {
913 this_type& ep;
914 Packet packet;
915 bool from_queue = false;
916 error_code last_ec = error_code{};
917 enum { dispatch, write, bind, complete } state = dispatch;
918
919 template <typename Self>
920 void operator()(
921 Self& self,
922 error_code const& ec = error_code{},
923 std::size_t /*bytes_transferred*/ = 0
924 ) {
925 if (ec) {
926 ASYNC_MQTT_LOG("mqtt_impl", info)
927 << ASYNC_MQTT_ADD_VALUE(address, &ep)
928 << "send error:" << ec.message();
929 last_ec = ec;
930 state = complete;
931 bind_dispatch(force_move(self));
932 return;
933 }
934
935 switch (state) {
936 case dispatch: {
937 state = write;
938 auto& a_ep{ep};
939 as::dispatch(
940 as::bind_executor(
941 a_ep.stream_->raw_strand(),
942 force_move(self)
943 )
944 );
945 } break;
946 case write: {
947 BOOST_ASSERT(ep.in_strand());
948 state = bind;
949 if constexpr(
950 std::is_same_v<std::decay_t<Packet>, basic_packet_variant<PacketIdBytes>> ||
951 std::is_same_v<std::decay_t<Packet>, basic_store_packet_variant<PacketIdBytes>>
952 ) {
953 packet.visit(
954 overload {
955 [&](auto& actual_packet) {
956 if (process_send_packet(self, actual_packet)) {
957 auto& a_ep{ep};
958 a_ep.stream_->write_packet(
959 force_move(actual_packet),
960 force_move(self)
961 );
962 if constexpr(is_connack<std::remove_reference_t<decltype(actual_packet)>>()) {
963 // server send stored packets after connack sent
964 a_ep.send_stored();
965 }
966 if constexpr(Role == role::client) {
967 a_ep.reset_pingreq_send_timer();
968 }
969 }
970 },
971 [&](system_error&) {}
972 }
973 );
974 }
975 else {
976 if (process_send_packet(self, packet)) {
977 auto& a_ep{ep};
978 auto& a_packet{packet};
979 a_ep.stream_->write_packet(
980 force_move(a_packet),
981 force_move(self)
982 );
983 if constexpr(is_connack<Packet>()) {
984 // server send stored packets after connack sent
985 a_ep.send_stored();
986 }
987 if constexpr(Role == role::client) {
988 a_ep.reset_pingreq_send_timer();
989 }
990 }
991 }
992 } break;
993 case bind: {
994 BOOST_ASSERT(ep.in_strand());
995 last_ec = ec;
996 state = complete;
997 bind_dispatch(force_move(self));
998 } break;
999 case complete:
1000 // out of strand
1001 self.complete(last_ec);
1002 break;
1003 }
1004 }

1006 template <typename Self, typename ActualPacket>
1007 bool process_send_packet(Self& self, ActualPacket& actual_packet) {
1008 // MQTT protocol sendable packet check
1009 if (
1010 !(
1011 (can_send_as_client(Role) && is_client_sendable<std::decay_t<ActualPacket>>()) ||
1012 (can_send_as_server(Role) && is_server_sendable<std::decay_t<ActualPacket>>())
1013 )
1014 ) {
1015 self.complete(
1016 make_error(
1017 errc::protocol_error,
1018 "packet cannot be send by MQTT protocol"
1019 )
1020 );
1021 return false;
1022 }
1023
1024 auto version_check =
1025 [&] {
1026 if (ep.protocol_version_ == protocol_version::v3_1_1 && is_v3_1_1<ActualPacket>()) {
1027 return true;
1028 }
1029 if (ep.protocol_version_ == protocol_version::v5 && is_v5<ActualPacket>()) {
1030 return true;
1031 }
1032 return false;
1033 };
1034
1035 // connection status check
1036 if constexpr(is_connect<ActualPacket>()) {
1037 if (ep.status_ != connection_status::closed) {
1038 self.complete(
1039 make_error(
1040 errc::protocol_error,
1041 "connect_packet can only be send on connection_status::closed"
1042 )
1043 );
1044 return false;
1045 }
1046 if (!version_check()) {
1047 self.complete(
1048 make_error(
1049 errc::protocol_error,
1050 "protocol version mismatch"
1051 )
1052 );
1053 return false;
1054 }
1055 }
1056 else if constexpr(is_connack<ActualPacket>()) {
1057 if (ep.status_ != connection_status::connecting) {
1058 self.complete(
1059 make_error(
1060 errc::protocol_error,
1061 "connack_packet can only be send on connection_status::connecting"
1062 )
