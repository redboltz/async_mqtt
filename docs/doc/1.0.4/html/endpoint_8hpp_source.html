782 template <typename Packet>
783 struct send_impl {
784 this_type& ep;
785 Packet packet;
786 bool from_queue = false;
787 enum { dispatch, write, complete } state = dispatch;
788
789 template <typename Self>
790 void operator()(
791 Self& self,
792 error_code const& ec = error_code{},
793 std::size_t /*bytes_transferred*/ = 0
794 ) {
795 if (ec) {
796 ASYNC_MQTT_LOG("mqtt_impl", info)
797 << ASYNC_MQTT_ADD_VALUE(address, &ep)
798 << "send error:" << ec.message();
799 self.complete(ec);
800 return;
801 }
802
803 switch (state) {
804 case dispatch: {
805 state = write;
806 auto& a_ep{ep};
807 as::dispatch(
808 a_ep.strand(),
809 force_move(self)
810 );
811 } break;
812 case write: {
813 BOOST_ASSERT(ep.strand().running_in_this_thread());
814 state = complete;
815 if constexpr(std::is_same_v<std::decay_t<Packet>, basic_packet_variant<PacketIdBytes>>) {
816 packet.visit(
817 overload {
818 [&](auto& actual_packet) {
819 if (process_send_packet(self, actual_packet)) {
820 auto& a_ep{ep};
821 a_ep.stream_->write_packet(
822 force_move(actual_packet),
823 force_move(self)
824 );
825 if constexpr(is_connack<std::remove_reference_t<decltype(actual_packet)>>()) {
826 // server send connack after connack sent
827 a_ep.send_stored();
828 }
829 if constexpr(Role == role::client) {
830 a_ep.reset_pingreq_send_timer();
831 }
832 }
833 },
834 [&](system_error&) {}
835 }
836 );
837 }
838 else {
839 if (process_send_packet(self, packet)) {
840 auto& a_ep{ep};
841 auto& a_packet{packet};
842 a_ep.stream_->write_packet(
843 force_move(a_packet),
844 force_move(self)
845 );
846 if constexpr(is_connack<Packet>()) {
847 // server send connack after connack sent
848 a_ep.send_stored();
849 }
850 if constexpr(Role == role::client) {
851 a_ep.reset_pingreq_send_timer();
852 }
853 }
854 }
855 } break;
856 case complete:
857 BOOST_ASSERT(ep.strand().running_in_this_thread());
858 self.complete(ec);
859 break;
860 }

863 template <typename Self, typename ActualPacket>
864 bool process_send_packet(Self& self, ActualPacket& actual_packet) {
865 // MQTT protocol sendable packet check
866 if ((can_send_as_client(Role) && !is_client_sendable<std::decay_t<ActualPacket>>()) ||
867 (can_send_as_server(Role) && !is_server_sendable<std::decay_t<ActualPacket>>())
868 ) {
869 self.complete(
870 make_error(
871 errc::protocol_error,
872 "Packet cannot be send by MQTT protocol"
873 )
874 );
875 return false;
876 }
877
878 // connection status check
879 if constexpr(is_connect<ActualPacket>()) {
880 if (ep.status_ != connection_status::disconnected) {
881 self.complete(
882 make_error(
883 errc::protocol_error,
884 "connect_packet can only be send on connection_status::disconnected"
885 )
886 );
887 return false;
888 }
889 }
890 else if constexpr(is_connack<ActualPacket>()) {
891 if (ep.status_ != connection_status::connecting) {
892 self.complete(
893 make_error(
894 errc::protocol_error,
895 "connack_packet can only be send on connection_status::connecting"
896 )
897 );
898 return false;
899 }
900 }
901 else if constexpr(std::is_same_v<v5::auth_packet, Packet>) {
902 if (ep.status_ != connection_status::connected &&
903 ep.status_ != connection_status::connecting) {
904 self.complete(
905 make_error(
906 errc::protocol_error,
907 "auth packet can only be send on connection_status::connecting or status::connected"
908 )
909 );
910 return false;
911 }
912 }
913 else {
914 if (ep.status_ != connection_status::connected) {
915 if constexpr(!is_publish<std::decay_t<ActualPacket>>()) {
916 self.complete(
917 make_error(
918 errc::protocol_error,
919 "packet can only be send on connection_status::connected"
920 )
921 );
922 return false;
923 }
924 }
925 }
926
927 // sending process
928 bool topic_alias_validated = false;
929
930 if constexpr(std::is_same_v<v3_1_1::connect_packet, std::decay_t<ActualPacket>>) {
931 ep.initialize();
932 ep.status_ = connection_status::connecting;
933 auto keep_alive = actual_packet.keep_alive();
934 if (keep_alive != 0 && !ep.pingreq_send_interval_ms_) {
935 ep.pingreq_send_interval_ms_.emplace(keep_alive * 1000);
936 }
937 if (actual_packet.clean_session()) {
938 ep.pid_man_.clear();
939 ep.store_.clear();
940 ep.need_store_ = false;
941 }
942 else {
943 ep.need_store_ = true;
944 }
945 ep.topic_alias_send_ = nullopt;
946 }
947
948 if constexpr(std::is_same_v<v5::connect_packet, std::decay_t<ActualPacket>>) {
949 ep.initialize();
950 ep.status_ = connection_status::connecting;
951 auto keep_alive = actual_packet.keep_alive();
952 if (keep_alive != 0 && !ep.pingreq_send_interval_ms_) {
953 ep.pingreq_send_interval_ms_.emplace(keep_alive * 1000);
954 }
955 if (actual_packet.clean_start()) {
956 ep.pid_man_.clear();
957 ep.store_.clear();
958 }
959 for (auto const& prop : actual_packet.props()) {
960 prop.visit(
961 overload {
962 [&](property::topic_alias_maximum const& p) {
963 if (p.val() != 0) {
964 ep.topic_alias_recv_.emplace(p.val());
965 }
966 },
967 [&](property::receive_maximum const& p) {
968 BOOST_ASSERT(p.val() != 0);
969 ep.publish_recv_max_ = p.val();
970 },
971 [&](property::maximum_packet_size const& p) {
972 BOOST_ASSERT(p.val() != 0);
973 ep.maximum_packet_size_recv_ = p.val();
974 },
975 [&](property::session_expiry_interval const& p) {
976 if (p.val() != 0) {
977 ep.need_store_ = true;
978 }
979 },
980 [](auto const&){}
981 }
982 );
983 }
984 }
985
986 if constexpr(std::is_same_v<v3_1_1::connack_packet, std::decay_t<ActualPacket>>) {
987 if (actual_packet.code() == connect_return_code::accepted) {
988 ep.status_ = connection_status::connected;
989 }
990 else {
991 ep.status_ = connection_status::disconnecting;
992 }
993 }
994
995 if constexpr(std::is_same_v<v5::connack_packet, std::decay_t<ActualPacket>>) {
996 if (actual_packet.code() == connect_reason_code::success) {
997 ep.status_ = connection_status::connected;
998 for (auto const& prop : actual_packet.props()) {
999 prop.visit(
1000 overload {
1001 [&](property::topic_alias_maximum const& p) {
1002 if (p.val() != 0) {
1003 ep.topic_alias_recv_.emplace(p.val());
1004 }
1005 },
1006 [&](property::receive_maximum const& p) {
1007 BOOST_ASSERT(p.val() != 0);
1008 ep.publish_recv_max_ = p.val();
1009 },
1010 [&](property::maximum_packet_size const& p) {
1011 BOOST_ASSERT(p.val() != 0);
1012 ep.maximum_packet_size_recv_ = p.val();
1013 },
1014 [](auto const&){}
1015 }
1016 );
1017 }
1018 }
1019 else {
1020 ep.status_ = connection_status::disconnecting;
1021 }
1022 }

1024 // store publish/pubrel packet
1025 if constexpr(is_publish<std::decay_t<ActualPacket>>()) {
1026 if (actual_packet.opts().get_qos() == qos::at_least_once ||
1027 actual_packet.opts().get_qos() == qos::exactly_once
1028 ) {
1029 BOOST_ASSERT(ep.pid_man_.is_used_id(actual_packet.packet_id()));
1030 if (ep.need_store_) {
1031 if constexpr(is_instance_of<v5::basic_publish_packet, std::decay_t<ActualPacket>>::value) {
1032 auto ta_opt = get_topic_alias(actual_packet.props());
1033 if (actual_packet.topic().empty()) {
1034 auto topic_opt = validate_topic_alias(self, ta_opt);
1035 if (!topic_opt) {
1036 auto packet_id = actual_packet.packet_id();
1037 if (packet_id != 0) {
1038 ep.pid_man_.release_id(packet_id);
1039 }
1040 return false;
1041 }
1042 topic_alias_validated = true;
1043 auto props = actual_packet.props();
1044 auto it = props.cbegin();
1045 auto end = props.cend();
1046 for (; it != end; ++it) {
1047 if (it->id() == property::id::topic_alias) {
1048 props.erase(it);
1049 break;
1050 }
1051 }
1052
1053 auto store_packet =
1054 ActualPacket(
1055 actual_packet.packet_id(),
1056 allocate_buffer(*topic_opt),
1057 actual_packet.payload(),
1058 actual_packet.opts(),
1059 force_move(props)
1060 );
1061 if (!validate_maximum_packet_size(self, store_packet)) {
1062 auto packet_id = actual_packet.packet_id();
1063 if (packet_id != 0) {
1064 ep.pid_man_.release_id(packet_id);
1065 }
1066 return false;
1067 }
1068 // add new packet that doesn't have topic_aliass to store
1069 // the original packet still use topic alias to send
1070 store_packet.set_dup(true);
1071 ep.store_.add(force_move(store_packet));
1072 }
1073 else {
1074 auto props = actual_packet.props();
1075 auto it = props.cbegin();
1076 auto end = props.cend();
1077 for (; it != end; ++it) {
1078 if (it->id() == property::id::topic_alias) {
1079 props.erase(it);
1080 break;
1081 }
1082 }
1083
1084 auto store_packet =
1085 ActualPacket(
1086 actual_packet.packet_id(),
1087 actual_packet.topic(),
1088 actual_packet.payload(),
1089 actual_packet.opts(),
1090 force_move(props)
1091 );
1092 if (!validate_maximum_packet_size(self, store_packet)) {
1093 auto packet_id = actual_packet.packet_id();
1094 if (packet_id != 0) {
1095 ep.pid_man_.release_id(packet_id);
1096 }
1097 return false;
1098 }
1099 store_packet.set_dup(true);
1100 ep.store_.add(force_move(store_packet));
1101 }
1102 }
1103 else {
1104 if (!validate_maximum_packet_size(self, actual_packet)) {
1105 auto packet_id = actual_packet.packet_id();
1106 if (packet_id != 0) {
1107 ep.pid_man_.release_id(packet_id);
1108 }
1109 return false;
1110 }
1111 auto store_packet{actual_packet};
1112 store_packet.set_dup(true);
1113 ep.store_.add(force_move(store_packet));
1114 }
1115 }
1116 if (actual_packet.opts().get_qos() == qos::exactly_once) {
1117 ep.qos2_publish_processing_.insert(actual_packet.packet_id());
1118 }
1119 }
1120 }
1121
1122 if constexpr(is_instance_of<v5::basic_publish_packet, std::decay_t<ActualPacket>>::value) {
1123 // apply topic_alias
1124 auto ta_opt = get_topic_alias(actual_packet.props());
1125 if (actual_packet.topic().empty()) {
1126 if (!topic_alias_validated &&
1127 !validate_topic_alias(self, ta_opt)) {
1128 auto packet_id = actual_packet.packet_id();
1129 if (packet_id != 0) {
1130 ep.pid_man_.release_id(packet_id);
1131 }
1132 return false;
1133 }
1134 // use topic_alias set by user
1135 }
1136 else {
1137 if (ta_opt) {
1138 if (validate_topic_alias_range(self, *ta_opt)) {
1139 ASYNC_MQTT_LOG("mqtt_impl", trace)
1140 << ASYNC_MQTT_ADD_VALUE(address, &ep)
1141 << "topia alias : "
1142 << actual_packet.topic() << " - " << *ta_opt
1143 << " is registered." ;
1144 ep.topic_alias_send_->insert_or_update(actual_packet.topic(), *ta_opt);
1145 }
1146 else {
1147 auto packet_id = actual_packet.packet_id();
1148 if (packet_id != 0) {
1149 ep.pid_man_.release_id(packet_id);
1150 }
1151 return false;
1152 }
1153 }
1154 else if (ep.auto_map_topic_alias_send_) {
1155 if (ep.topic_alias_send_) {
1156 if (auto ta_opt = ep.topic_alias_send_->find(actual_packet.topic())) {
1157 ASYNC_MQTT_LOG("mqtt_impl", trace)
1158 << ASYNC_MQTT_ADD_VALUE(address, &ep)
1159 << "topia alias : " << actual_packet.topic() << " - " << *ta_opt
1160 << " is found." ;
1161 actual_packet.remove_topic_add_topic_alias(*ta_opt);
1162 }
1163 else {
1164 auto lru_ta = ep.topic_alias_send_->get_lru_alias();
1165 ep.topic_alias_send_->insert_or_update(actual_packet.topic(), lru_ta); // remap topic alias
1166 actual_packet.add_topic_alias(lru_ta);
1167 }
1168 }
1169 }
1170 else if (ep.auto_replace_topic_alias_send_) {
1171 if (ep.topic_alias_send_) {
1172 if (auto ta_opt = ep.topic_alias_send_->find(actual_packet.topic())) {
1173 ASYNC_MQTT_LOG("mqtt_impl", trace)
1174 << ASYNC_MQTT_ADD_VALUE(address, &ep)
1175 << "topia alias : " << actual_packet.topic() << " - " << *ta_opt
1176 << " is found." ;
1177 actual_packet.remove_topic_add_topic_alias(*ta_opt);
1178 }
1179 }
1180 }
1181 }
1182
1183 // receive_maximum for sending
1184 if (!from_queue && ep.enqueue_publish(actual_packet)) {
1185 self.complete(
1186 make_error(
1187 errc::success,
1188 "publish_packet is enqueued due to receive_maximum for sending"
1189 )
1190 );
1191 return false;
1192 }
1193 }
1194
1195 if constexpr(is_instance_of<v5::basic_puback_packet, std::decay_t<ActualPacket>>::value) {
1196 ep.publish_recv_.erase(actual_packet.packet_id());
1197 }
1198
1199
1200 if constexpr(is_instance_of<v5::basic_pubrec_packet, std::decay_t<ActualPacket>>::value) {
1201 if (is_error(actual_packet.code())) {
1202 ep.publish_recv_.erase(actual_packet.packet_id());
1203 ep.qos2_publish_handled_.erase(actual_packet.packet_id());
1204 }
1205 }
1206
1207 if constexpr(is_pubrel<std::decay_t<ActualPacket>>()) {
1208 BOOST_ASSERT(ep.pid_man_.is_used_id(actual_packet.packet_id()));
1209 if (ep.need_store_) ep.store_.add(actual_packet);
1210 }
1211
1212 if constexpr(is_instance_of<v5::basic_pubcomp_packet, std::decay_t<ActualPacket>>::value) {
1213 ep.publish_recv_.erase(actual_packet.packet_id());
1214 }
1215
1216 if constexpr(is_subscribe<std::decay_t<ActualPacket>>()) {
1217 BOOST_ASSERT(ep.pid_man_.is_used_id(actual_packet.packet_id()));
1218 }
1219
1220 if constexpr(is_unsubscribe<std::decay_t<ActualPacket>>()) {
1221 BOOST_ASSERT(ep.pid_man_.is_used_id(actual_packet.packet_id()));
1222 }
1223
1224 if constexpr(is_pingreq<std::decay_t<ActualPacket>>()) {
1225 ep.reset_pingresp_recv_timer();
1226 }
1227
1228 if constexpr(is_disconnect<std::decay_t<ActualPacket>>()) {
1229 ep.status_ = connection_status::disconnecting;
1230 }
1231
1232 if (!validate_maximum_packet_size(self, actual_packet)) {
1233 if constexpr(own_packet_id<std::decay_t<ActualPacket>>()) {
1234 auto packet_id = actual_packet.packet_id();
1235 if (packet_id != 0) {
1236 ep.pid_man_.release_id(packet_id);
1237 }
1238 }
1239 return false;
1240 }
1241
1242 if constexpr(is_publish<std::decay_t<ActualPacket>>()) {
1243 if (ep.status_ != connection_status::connected) {
1244 // offline publish
1245 self.complete(
1246 make_error(
1247 errc::success,
1248 "packet is stored but not sent"
1249 )
1250 );
1251 return false;
1252 }
1253 }
1254
1255 return true;
1256 }

1329 struct recv_impl {
1330 this_type& ep;
1331 optional<system_error> decided_error = nullopt;
1332 enum { initiate, disconnect, close, complete } state = initiate;
1333
1334 template <typename Self>
1335 void operator()(
1336 Self& self,
1337 error_code const& ec = error_code{},
1338 buffer buf = buffer{}
1339 ) {
1340 if (ec) {
1341 ASYNC_MQTT_LOG("mqtt_impl", info)
1342 << ASYNC_MQTT_ADD_VALUE(address, &ep)
1343 << "recv error:" << ec.message();
1344 decided_error.emplace(ec);
1345 ep.status_ = connection_status::disconnected;
1346 ep.recv_processing_ = false;
1347 state = close;
1348 auto& a_ep{ep};
1349 a_ep.close(force_move(self));
1350 return;
1351 }
1352
1353 switch (state) {
1354 case initiate: {
1355 state = complete;
1356 auto& a_ep{ep};
1357 a_ep.stream_->read_packet(force_move(self));
1358 } break;
1359 case complete: {
1360 BOOST_ASSERT(ep.strand().running_in_this_thread());
1361 if (buf.size() > ep.maximum_packet_size_recv_) {
1362 // on v3.1.1 maximum_packet_size_recv_ is initialized as packet_size_no_limit
1363 BOOST_ASSERT(ep.protocol_version_ == protocol_version::v5);
1364 state = disconnect;
1365 decided_error.emplace(
1366 make_error(
1367 errc::bad_message,
1368 "too large packet received"
1369 )
1370 );
1371 auto& a_ep{ep};
1372 a_ep.send(
1373 v5::disconnect_packet{
1374 disconnect_reason_code::packet_too_large
1375 },
1376 force_move(self)
1377 );
1378 return;
1379 }
1380
1381 auto v = buffer_to_basic_packet_variant<PacketIdBytes>(buf, ep.protocol_version_);
1382 ASYNC_MQTT_LOG("mqtt_impl", trace)
1383 << ASYNC_MQTT_ADD_VALUE(address, &ep)
1384 << "recv:" << v;
1385 bool call_complete = true;
1386 v.visit(
1387 // do internal protocol processing
1388 overload {
1389 [&](v3_1_1::connect_packet& p) {
1390 ep.initialize();
1391 ep.protocol_version_ = protocol_version::v3_1_1;
1392 ep.status_ = connection_status::connecting;
1393 auto keep_alive = p.keep_alive();
1394 if (keep_alive != 0) {
1395 ep.pingreq_recv_timeout_ms_.emplace(keep_alive * 1000 * 3 / 2);
1396 }
1397 if (p.clean_session()) {
1398 ep.need_store_ = false;
1399 }
1400 else {
1401 ep.need_store_ = true;
1402 }
1403 },
1404 [&](v5::connect_packet& p) {
1405 ep.initialize();
1406 ep.protocol_version_ = protocol_version::v5;
1407 ep.status_ = connection_status::connecting;
1408 auto keep_alive = p.keep_alive();
1409 if (keep_alive != 0) {
1410 ep.pingreq_recv_timeout_ms_.emplace(keep_alive * 1000 * 3 / 2);
1411 }
1412 for (auto const& prop : p.props()) {
1413 prop.visit(
1414 overload {
1415 [&](property::topic_alias_maximum const& p) {
1416 if (p.val() > 0) {
1417 ep.topic_alias_send_.emplace(p.val());
1418 }
1419 },
1420 [&](property::receive_maximum const& p) {
1421 BOOST_ASSERT(p.val() != 0);
1422 ep.publish_send_max_ = p.val();
1423 },
1424 [&](property::maximum_packet_size const& p) {
1425 BOOST_ASSERT(p.val() != 0);
1426 ep.maximum_packet_size_send_ = p.val();
1427 },
1428 [&](property::session_expiry_interval const& p) {
1429 if (p.val() != 0) {
1430 ep.need_store_ = true;
1431 }
1432 },
1433 [](auto const&) {
1434 }
1435 }
1436 );
1437 }
1438 },
1439 [&](v3_1_1::connack_packet& p) {
1440 if (p.code() == connect_return_code::accepted) {
1441 ep.status_ = connection_status::connected;
1442 if (p.session_present()) {
1443 ep.send_stored();
1444 }
1445 else {
1446 ep.pid_man_.clear();
1447 ep.store_.clear();
1448 }
1449 }
1450 },
1451 [&](v5::connack_packet& p) {
1452 if (p.code() == connect_reason_code::success) {
1453 ep.status_ = connection_status::connected;
1454
1455 for (auto const& prop : p.props()) {
1456 prop.visit(
1457 overload {
1458 [&](property::topic_alias_maximum const& p) {
1459 if (p.val() > 0) {
1460 ep.topic_alias_send_.emplace(p.val());
1461 }
1462 },
1463 [&](property::receive_maximum const& p) {
1464 BOOST_ASSERT(p.val() != 0);
1465 ep.publish_send_max_ = p.val();
1466 },
1467 [&](property::maximum_packet_size const& p) {
1468 BOOST_ASSERT(p.val() != 0);
1469 ep.maximum_packet_size_send_ = p.val();
1470 },
1471 [](auto const&) {
1472 }
1473 }
1474 );
1475 }
1476
1477 if (p.session_present()) {
1478 ep.send_stored();
1479 }
1480 else {
1481 ep.pid_man_.clear();
1482 ep.store_.clear();
1483 }
1484 }
1485 },
1486 [&](v3_1_1::basic_publish_packet<PacketIdBytes>& p) {

1488 case qos::at_least_once: {

1491 v3_1_1::basic_puback_packet<PacketIdBytes>(p.packet_id()),
1492 [](system_error const&){}

1496 case qos::exactly_once:
1497 call_complete = process_qos2_publish(self, protocol_version::v3_1_1, p.packet_id());

1505 case qos::at_least_once: {

1544 disconnect_reason_code::receive_maximum_exceeded

1582 ASYNC_MQTT_LOG("mqtt_impl", error)
1583 << ASYNC_MQTT_ADD_VALUE(address, &ep)
1584 << "no matching topic alias: "
1585 << *ta_opt;
1586 state = disconnect;
1587 decided_error.emplace(
1588 make_error(
1589 errc::bad_message,
1590 "topic alias invalid"
1591 )
1592 );
1593 auto& a_ep{ep};
1594 a_ep.send(
1595 v5::disconnect_packet{
1596 disconnect_reason_code::topic_alias_invalid
1597 },
1598 force_move(self)
1599 );
1600 return;

1603 p.add_topic(allocate_buffer(topic));

1605 }
1606 }
1607 else {
1608 ASYNC_MQTT_LOG("mqtt_impl", error)
1609 << ASYNC_MQTT_ADD_VALUE(address, &ep)

1612 decided_error.emplace(
1613 make_error(
1614 errc::bad_message,
1615 "topic alias invalid"
1616 )
1617 );

1621 disconnect_reason_code::topic_alias_invalid
1622 },
1623 force_move(self)
1624 );
1625 return;
1626 }
1627 }

1629 if (auto ta_opt = get_topic_alias(p.props())) {

1631 *ta_opt > ep.topic_alias_recv_->max()) {
1632 state = disconnect;
1633 decided_error.emplace(
1634 make_error(
1635 errc::bad_message,

1668 },

1684 ep.qos2_publish_processing_.erase(packet_id);
1685 --ep.publish_send_count_;
