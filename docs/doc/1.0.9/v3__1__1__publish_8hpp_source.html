75 packet_id_t packet_id,
76 buffer topic_name,
77 BufferSequence payloads,
78 pub::opts pubopts

81 make_fixed_header(control_packet_type::publish, 0b0000) | std::uint8_t(pubopts)

83 topic_name_{force_move(topic_name)},
84 packet_id_(PacketIdBytes),
85 remaining_length_(
86 2 // topic name length
87 + topic_name_.size() // topic name
88 + ( (pubopts.get_qos() == qos::at_least_once || pubopts.get_qos() == qos::exactly_once)
89 ? PacketIdBytes // packet_id

96 topic_name_length_buf_.data()
97 );
98 auto b = buffer_sequence_begin(payloads);
99 auto e = buffer_sequence_end(payloads);
100 auto num_of_payloads = static_cast<std::size_t>(std::distance(b, e));
101 payloads_.reserve(num_of_payloads);
102 for (; b != e; ++b) {
103 auto const& payload = *b;
104 remaining_length_ += payload.size();
105 payloads_.push_back(payload);
106 }
107#if 0 // TBD
108 utf8string_check(topic_name_);
109#endif
110 auto rb = val_to_variable_bytes(boost::numeric_cast<std::uint32_t>(remaining_length_));
111 for (auto e : rb) {
112 remaining_length_buf_.push_back(e);
113 }
114 switch (pubopts.get_qos()) {
115 case qos::at_most_once:
116 if (packet_id != 0) {
117 throw make_error(
118 errc::bad_message,
119 "v3_1_1::publish_packet qos0 but non 0 packet_id"
120 );
121 }
122 endian_store(0, packet_id_.data());
123 break;
124 case qos::at_least_once:
125 case qos::exactly_once:
126 if (packet_id == 0) {
127 throw make_error(
128 errc::bad_message,
129 "v3_1_1::publish_packet qos not 0 but packet_id is 0"
130 );
131 }
132 endian_store(packet_id, packet_id_.data());
133 break;
134 default:
135 throw make_error(
136 errc::bad_message,
137 "v3_1_1::publish_packet qos is invalid"
138 );
139 break;

170 basic_publish_packet(buffer buf)
171 : packet_id_(PacketIdBytes) {
172 // fixed_header
173 if (buf.empty()) {
174 throw make_error(
175 errc::bad_message,
176 "v3_1_1::publish_packet fixed_header doesn't exist"
177 );
178 }
179 fixed_header_ = static_cast<std::uint8_t>(buf.front());
180 auto qos_value = pub::get_qos(fixed_header_);
181 buf.remove_prefix(1);
182 auto cpt_opt = get_control_packet_type_with_check(static_cast<std::uint8_t>(fixed_header_));
183 if (!cpt_opt || *cpt_opt != control_packet_type::publish) {
184 throw make_error(
185 errc::bad_message,
186 "v3_1_1::publish_packet fixed_header is invalid"
187 );
188 }
189
190 // remaining_length
191 if (auto vl_opt = insert_advance_variable_length(buf, remaining_length_buf_)) {
192 remaining_length_ = *vl_opt;
193 }
194 else {
195 throw make_error(errc::bad_message, "v3_1_1::publish_packet remaining length is invalid");
196 }
197 if (remaining_length_ != buf.size()) {
198 throw make_error(errc::bad_message, "v3_1_1::publish_packet remaining length doesn't match buf.size()");
199 }
200
201 // topic_name_length
202 if (!insert_advance(buf, topic_name_length_buf_)) {
203 throw make_error(
204 errc::bad_message,
205 "v3_1_1::publish_packet length of topic_name is invalid"
206 );
207 }
208 auto topic_name_length = endian_load<std::uint16_t>(topic_name_length_buf_.data());
209
210 // topic_name
211 if (buf.size() < topic_name_length) {
212 throw make_error(
213 errc::bad_message,
214 "v3_1_1::publish_packet topic_name doesn't match its length"
215 );
216 }
217 topic_name_ = buf.substr(0, topic_name_length);
218#if 0 // TBD
219 utf8string_check(topic_name_);
220#endif
221 buf.remove_prefix(topic_name_length);
222
223 // packet_id
224 switch (qos_value) {
225 case qos::at_most_once:
226 endian_store(packet_id_t{0}, packet_id_.data());
227 break;
228 case qos::at_least_once:
229 case qos::exactly_once:
230 if (!copy_advance(buf, packet_id_)) {
231 throw make_error(
232 errc::bad_message,
233 "v3_1_1::publish_packet packet_id doesn't exist"
234 );
235 }
236 break;
237 default:
238 throw make_error(
239 errc::bad_message,
240 "v3_1_1::publish_packet qos is invalid"
241 );
242 break;
243 };
244
245 // payload
246 if (!buf.empty()) {
247 payloads_.emplace_back(force_move(buf));
248 }
249 }

260 std::vector<as::const_buffer> const_buffer_sequence() const {
261 std::vector<as::const_buffer> ret;
262 ret.reserve(num_of_const_buffer_sequence());
263 ret.emplace_back(as::buffer(&fixed_header_, 1));
264 ret.emplace_back(as::buffer(remaining_length_buf_.data(), remaining_length_buf_.size()));
265 ret.emplace_back(as::buffer(topic_name_length_buf_.data(), topic_name_length_buf_.size()));
266 ret.emplace_back(as::buffer(topic_name_));
267 if (packet_id() != 0) {
268 ret.emplace_back(as::buffer(packet_id_.data(), packet_id_.size()));
269 }
270 for (auto const& payload : payloads_) {
271 ret.emplace_back(as::buffer(payload));
272 }
273 return ret;
274 }
